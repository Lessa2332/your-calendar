// Спрощений запуск MindAR function startMindAR() { const sceneEl = document.querySelector('a-scene'); if (!sceneEl) { setTimeout(startMindAR, 300); return; } // Чекаємо на системи if (!sceneEl.systems) { setTimeout(startMindAR, 300); return; } const mindarSystem = sceneEl.systems['mindar-image-system'] || sceneEl.systems['mindar-image']; if (mindarSystem && typeof mindarSystem.start === 'function') { console.log('Starting MindAR system...'); try { mindarSystem.start(); document.getElementById('hint').style.display = 'none'; } catch (error) { console.warn('MindAR start failed, retrying...', error); setTimeout(startMindAR, 500); } } else { console.log('MindAR system not found, retrying...'); setTimeout(startMindAR, 300); } }
